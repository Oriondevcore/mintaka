let report \= {
    scannedFolder: targetFolder,
    totalFiles: 0,
    duplicates: {},
    versionedFiles: \[\],
    largeFiles: \[\],
    projectFolders: \[\],
    extensions: {}
};

const versionRegex \= /(v\\d+|version\\d+|final|copy)/i;

function scanDirectory(dir) {

    const items \= fs.readdirSync(dir);

    items.forEach(item \=\> {

        const fullPath \= path.join(dir, item);

        let stats;

        try {
            stats \= fs.statSync(fullPath);
        } catch {
            return;
        }

        if (stats.isDirectory()) {

            if (
                fs.existsSync(path.join(fullPath, "package.json")) ||
                fs.existsSync(path.join(fullPath, ".git")) ||
                fs.existsSync(path.join(fullPath, "node\_modules"))
            ) {
                report.projectFolders.push(fullPath);
            }

            scanDirectory(fullPath);

        } else {

            report.totalFiles++;

            const ext \= path.extname(item).toLowerCase();

            if (\!report.extensions\[ext\]) {
                report.extensions\[ext\] \= 0;
            }

            report.extensions\[ext\]++;

            if (\!report.duplicates\[item\]) {
                report.duplicates\[item\] \= \[\];
            }

            report.duplicates\[item\].push(fullPath);

            if (versionRegex.test(item)) {
                report.versionedFiles.push(fullPath);
            }

            const sizeMB \= stats.size / (1024 \* 1024);

            if (sizeMB \> 50\) {
                report.largeFiles.push({
                    file: fullPath,
                    sizeMB: sizeMB.toFixed(2)
                });
            }

        }

    });

}

function scan(dir) {

  let items;

  try {
    items \= fs.readdirSync(dir);
  } catch {
    return;
  }

  for (const item of items) {

    const full \= path.join(dir, item);

    let stat;

    try {
      stat \= fs.statSync(full);
    } catch {
      continue;
    }

    if (stat.isDirectory()) {

      if (fs.existsSync(path.join(full, "package.json"))) {

        knowledge.projects.push({
          name: item,
          path: full
        });

      }

      scan(full);

    } else {

      knowledge.files.push({
        name: item,
        path: full
      });

    }

  }

}

function ensure(dir) {
  if (\!fs.existsSync(dir)) {
    fs.mkdirSync(dir, { recursive: true });
  }
}

function scan(dir) {

  let items;

  try {
    items \= fs.readdirSync(dir);
  } catch {
    return;
  }

  for (const item of items) {

    const full \= path.join(dir, item);

    let stat;

    try {
      stat \= fs.statSync(full);
    } catch {
      continue;
    }

    if (stat.isDirectory()) {

      if (item \=== "node\_modules") continue;

      scan(full);

    } else {

      const ext \= path.extname(item);

      if (TYPES\[ext\]) {

        const folder \= path.join(DEST, TYPES\[ext\]);

        ensure(folder);

        const dest \= path.join(folder, item);

        try {
          fs.copyFileSync(full, dest);
        } catch {}

      }

    }

  }
